    with engine.connect() as conn:

            {"key": header_key}

@app.get("/search/village")
def search_village(name: str, api_key: str = Depends(get_api_key)):
    with engine.begin() as conn:
        
        conn.execute(text("UPDATE api_users SET used_requests = used_requests + 1 WHERE api_key = :key"), {"key": api_key})
        
        
        query = text("""
            SELECT v.name, sd.name, d.name, s.name
            FROM villages v
            JOIN sub_districts sd ON v.sub_district_id = sd.id
            JOIN districts d ON sd.district_id = d.id
            JOIN states s ON d.state_id = s.id
            WHERE v.name ILIKE :vname LIMIT 10
        """)
